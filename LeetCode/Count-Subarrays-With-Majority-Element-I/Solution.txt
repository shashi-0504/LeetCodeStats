class Solution(object):
    def countMajoritySubarrays(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        n,ans=len(nums),0
        for i in range(n):
            c=0
            for j in range(i,n):
                c+=nums[j]==target
                if c>(j-i+1)//2:
                    ans+=1
        return ans
