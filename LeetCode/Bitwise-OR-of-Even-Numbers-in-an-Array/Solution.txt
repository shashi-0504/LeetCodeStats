class Solution(object):
    def evenNumberBitwiseORs(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        l=[]
        for i in nums:
            if i%2==0:
                l.append(i)
        if l:
            k=l[0]
            for i in range(1,len(l)):
                k=k|l[i]
            return k
        else:
            return 0
