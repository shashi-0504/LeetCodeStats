class Solution(object):
    def climbStairs(self, n, costs):
        """
        :type n: int
        :type costs: List[int]
        :rtype: int
        """
        l=[float("inf")]*(n+1)
        l[0]=0
        for i in range(1,n+1):
            cj=costs[i-1]
            for j in (1,2,3):
                if i-j>=0:
                    l[i]=min(l[i],l[i-j]+cj+(j*j))
        return l[n]
