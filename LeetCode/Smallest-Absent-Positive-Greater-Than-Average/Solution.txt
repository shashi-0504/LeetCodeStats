class Solution(object):
    def smallestAbsent(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        avg=sum(nums)/len(nums)
        ex=set(nums)
        c=int(avg)+1
        if c<=0:
            c=1
        while True:
            if c not in ex:
                return c
            c+=1
